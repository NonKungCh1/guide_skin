เพิ่มไอเทม

แอดออนของคุณใช้ ไอเทม 2 ตัว ทำงานคู่กัน:

ไอเทมหน้าที่slot
trigger itemถือในมือ กดคลิ้กขวาเพื่อ toggleมือถือปกติ
armor itemตัว armor จริง มีโมเดล 3D ติดslot.armor.legs

Armor Item (ไอเทมที่ใส่ช่อง armor จริง)

JSONBehaviorPack/items/myarmor.json
{
  "format_version": "1.21.10",
  "minecraft:item": {
    "description": {
      "identifier": "mynamespace:myarmor",  // namespace:ชื่อไอเทม
      "menu_category": {
        "category": "none"  // "none" = ไม่แสดงใน creative tab (ให้แจกด้วย command)
      }
    },
    "components": {
      "minecraft:icon": "myarmor_icon",   // ชื่อที่ register ใน item_texture.json
      "minecraft:max_stack_size": 1,
      "minecraft:display_name": {
        "value": "§lMyArmor"
      },
      "minecraft:wearable": {
        "slot": "slot.armor.legs"   // ช่องที่ใส่: head/chest/legs/feet
      }
    }
  }
}

Trigger Item (ไอเทมที่ถือกด toggle)

JSONBehaviorPack/items/mytrigger.json
{
  "format_version": "1.21.10",
  "minecraft:item": {
    "description": {
      "identifier": "mynamespace:mytrigger",
      "menu_category": { "category": "equipment" }
    },
    "components": {
      "minecraft:max_stack_size": 1,
      "minecraft:icon": "myarmor_icon",
      "minecraft:display_name": {
        "value": "§l§bMyArmor\n§r§7(กดคลิ้กขวา)"
      }
    }
  }
}

Molang Expression — อ่านค่าจากเกม

Molang คือภาษา expression ที่ Minecraft ใช้ในอนิเมชั่น — อ่านค่าจากเกมแบบ real-time ทุก frame

query.xxx — อ่านค่าจากเกม

queryความหมายตัวอย่างใช้งาน
query.target_y_rotationมุมหมุน Y ของกล้อง (ซ้าย/ขวา)ทำให้ skirt หันตามกล้อง
query.target_x_rotationมุมก้ม/เงยกล้องทำให้ hair ตกตามแรงโน้มถ่วง
query.is_sneaking1 ถ้ากำลัง sneak, 0 ถ้าไม่ก้มหลังเมื่อ sneak
query.is_moving1 ถ้ากำลังเดินเปิด walk animation
query.vertical_speedความเร็วแนวตั้ง (+ขึ้น/-ลง)เสื้อผ้าพริ้วตอนตกน้ำ
query.cape_flap_amountค่าสำหรับผ้าคลุม 0.0–1.0ผ้าพริ้วตามลม
query.modified_move_speedความเร็วเดิน 0.0–1.0เร่ง/ช้า animation ตามวิ่ง
query.head_y_rotation(0)มุม Y ของหัวหมุน hair ตามหัว

math.xxx — ฟังก์ชันคณิตศาสตร์

functionความหมาย
math.clamp(val, min, max)จำกัดค่าไม่ให้เกิน min/max
math.abs(val)ค่าสัมบูรณ์ (เอาเครื่องหมายลบออก)
math.lerp(a, b, t)เฉลี่ยระหว่าง a กับ b ตาม t (0–1)
math.sin(x)ไซน์ (องศา) — ทำ oscillation

อ่าน pre_animation จากแอดออนจริงของคุณ

นี่คือตัวอย่างที่ดึงมาจาก attachable ของแอดออน — อ่านทีละบรรทัด:

MOLANGpre_animation — คำนวณทุก frame ก่อน animation
// ① Smooth rotation — ค่อยๆ ขยับตามกล้อง Y (ไม่กระตุก)
variable.RAKVG > query.target_y_rotation ? {
  variable.RAKVG = variable.RAKVG
    - (math.abs(variable.RAKVG - query.target_y_rotation)
       / variable.RAKVGRAKVGRAKVG);  // หาร 8 = หน่วง 8 เท่า
};

// ② ลดมุมก้มเมื่อ sneak
variable.RAKVGRAKVG = variable.RAKVGRAKVG
  - (math.abs(query.target_y_rotation) / (45/2) + (query.is_sneaking * 2))
  + (query.is_sneaking * math.abs(query.target_y_rotation) / (45/2));
variable.RAKVGRAKVG = math.clamp(variable.RAKVGRAKVG, -45, 30);

// ③ Physics — ผ้าแกว่งตามการหมุนหัว (Spring simulation)
variable.XDHOQRAKVG = (
  variable.XDHOQRAKVG
  - (math.clamp(variable.KECCE..., -2.5, 2.5) * 0.025)
  - (variable.OUSZOQLALKBAFJS * 0.025)
) * 0.85;  // 0.85 = damping (แรงต้าน)

// ④ Cape animation — ผ้าพริ้วตามลม
variable.OUSZORAKVGBAFJSQBFSQRAKVG = math.clamp(
  math.lerp(0, -55, query.cape_flap_amount)
  - (variable.ALIBOALIBO * query.modified_move_speed),
  -70, 0
);

// ⑤ ค่าหมุนหัว Y แบบ offset
variable.QBFSQOXMCERAKVG = -query.target_y_rotation / variable.nine;
✅
สรุปการทำงานของ variable ทั้งหมด
variableค่ามาจากbone ที่ใช้
OUSZOQLALKBAFJSALIBOALIBO= 0 เกือบตลอด (neutralize)ทุก bone (= ไม่หมุน)
QBFSQOXMCERAKVG-target_y / 6head Z — หันตามกล้อง
RGBYMOXMCEYGKFGphysics ผ้าแกว่งAd, Ad1…Ad7 Z — ผ้าแกว่ง
KMFXWKMFXWspring จากการเดิน/กระโดดผ้าพริ้ว
OUSZORAKVGBAFJSQBFSQRAKVGcape_flap + move_speedผ้าคลุม/skirt

JSONResourcePack/attachables/myarmor.json
{
  "format_version": "1.10.0",
  "minecraft:attachable": {
    "description": {

      // ① ต้องตรงกับ identifier ในไฟล์ item
      "identifier": "mynamespace:myarmor",

      // ② เปิดใช้งานเมื่อผู้เล่น (ไม่ใช่ mob) ใส่ item นี้
      "item": {
        "mynamespace:myarmor":
          "query.owner_identifier == 'minecraft:player'"
      },

      // ③ Material — ควบคุมการ render
      "materials": {
        "default": "entity_emissive_alpha"
      },  // entity_alphatest = opacity แบบ hard | entity_emissive_alpha = เรืองแสง

      // ④ path ไปหาเทกเจอร์ (ไม่ใส่ .png)
      "textures": {
        "default": "textures/seana/myarmor"
      },

      // ⑤ ชื่อโมเดลต้องตรงกับ identifier ใน geometry.json
      "geometry": {
        "default": "geometry.myarmor"
      },

      "scripts": {
        // ⑥ initialize — ทำครั้งเดียวตอน entity โหลด
        "initialize": [
          "variable.OUSZOQLALKBAFJSALIBOALIBO=0;",
          "variable.ALIBOALIBO=13;",
          "variable.RAKVG=query.target_y_rotation;"
        ],

        // ⑦ pre_animation — คำนวณทุก frame ก่อน animation รัน
        "pre_animation": [
          "variable.QBFSQOXMCERAKVG=-query.target_y_rotation/variable.nine;",
          "variable.OUSZOOUOOM=(query.is_sneaking*-10);"
        ],

        // ⑧ animate — รัน animation ตามที่กำหนด
        "animate": [
          "MainAnimation"       // ชื่อ key ใน "animations" ด้านล่าง
        ]
      },

      // ⑨ map ชื่อ shorthand → ชื่อ animation จริงในไฟล์
      "animations": {
        "MainAnimation": "animation.myarmor"
      },

      "render_controllers": [
        "controller.render.armor"   // ใช้ vanilla render controller
      ]
    }
  }
}

Script (JavaScript)

JSBehaviorPack/scripts/main.js — ฉบับอ่านง่าย
import { world, system } from "@minecraft/server";

const TRIGGER = "mynamespace:mytrigger";
const ARMOR   = "mynamespace:myarmor";
const SLOT    = "slot.armor.legs";

// ─── ① auto-give เมื่อ spawn ───
world.afterEvents.playerSpawn.subscribe(async (ev) => {
  if (!ev.initialSpawn) return; // เฉพาะครั้งแรกที่เข้า world

  system.runTimeout(async () => {
    const p = ev.player;
    try {
      // ตรวจก่อนว่ายังไม่มี trigger item ค่อยแจก
      const r = await p.runCommandAsync(
        `execute unless entity @s[hasitem={item=${TRIGGER}}]`
        + ` run give @s ${TRIGGER} 1`
      );
      if (r.successCount > 0)
        p.sendMessage("§aได้รับ MyArmor แล้ว!");
    } catch {}
  }, 20); // delay 1 วินาที (20 ticks)
});

// ─── ② toggle ใส่/ถอด เมื่อกด trigger ───
world.beforeEvents.itemUse.subscribe((ev) => {
  const { itemStack: item, source: p } = ev;

  // กรองเฉพาะ trigger item ของเรา
  if (!p?.isValid() || !item || item.typeId !== TRIGGER) return;

  ev.cancel = true; // ยกเลิก action ปกติของ item

  system.run(async () => {
    try {
      // ตรวจว่า armor อยู่ใน slot แล้วหรือยัง
      const wearing = await p.runCommandAsync(
        `execute if entity @s[hasitem={item=${ARMOR},location=${SLOT}}]`
        + ` run clear @s ${ARMOR}`
      );

      if (wearing.successCount > 0) {
        // มีอยู่แล้ว → clear แล้วจบ (ถอดออก)
        return;
      }

      // ยังไม่มี → ใส่เข้าไปใน slot
      await p.runCommandAsync(`clear @s ${ARMOR}`); // ล้าง inventory ก่อน
      await p.runCommandAsync(
        `replaceitem entity @s ${SLOT} 0 ${ARMOR}`
      );
    } catch(e) {
      p.sendMessage("§cError: " + e.message);
    }
  });
});

Debug & ข้อผิดพลาดที่พบบ่อย

ปัญหาสาเหตุวิธีแก้
โมเดลไม่โชว์identifier ใน attachable ≠ itemตรวจสอบทุกตัวอักษร
เทกเจอร์ดำ/ขาวpath ผิดหรือขนาดไม่ถูกตรวจ path และขนาดเทกเจอร์
Script ไม่รันขาด module "script" ใน manifestเพิ่ม module type:"script"
อนิเมชั่นไม่ขยับชื่อ animation key ไม่ตรงกันตรวจ animations{} ใน attachable vs ไฟล์ animation
เกม crashJSON ผิด syntaxตรวจที่ jsonlint.com
UUID errorUUID ซ้ำกันGenerate UUID ใหม่ทั้งหมด
